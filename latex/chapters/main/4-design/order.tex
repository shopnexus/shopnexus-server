\subsection{API (Lớp xử lý)}
\label{subsec:order-apis}

\begin{table}[htbp]
\centering
\caption{Danh sách API --- module Mua hàng}

\toprule
\rowcolor{headerblue}
\thead{API} & \thead{Input chính} & \thead{Output} & \thead{Mô tả} \\
\midrule
GetCart          & AccountID & []CartItem & Lấy danh sách giỏ hàng \\
UpdateCart       & SkuID, Quantity/ DeltaQuantity & -- & Cập nhật số lượng; quantity=0 xoá item \\
ClearCart        & AccountID & -- & Xoá toàn bộ giỏ hàng \\
BuyerCheckout    & []CheckoutItem, BuyNow & []OrderItem & Tạo mục chờ xác nhận; BuyNow chỉ 1 item \\
\addlinespace
ListBuyerPending & AccountID, Status[], Page & Paginate[OrderItem] & DS mục chờ của buyer \\
CancelBuyerPending & AccountID, ItemID & -- & Buyer huỷ mục chờ \\
ListSellerPending & SellerID, Search, Page & Paginate[OrderItem] & DS mục chờ của seller \\
QuoteTransport   & ItemIDs, TransportOption & QuoteResult (cost) & Báo giá vận chuyển \\
ConfirmSellerPending & ItemIDs, TransportOption & Order & Seller xác nhận $\to$ tạo Order \\
RejectSellerPending & ItemIDs & -- & Seller từ chối \\
\addlinespace
GetBuyerOrder    & OrderID & Order & Chi tiết đơn (buyer) \\
GetSellerOrder   & OrderID & Order & Chi tiết đơn (seller) \\
ListBuyerConfirmed & BuyerID, Status[], Page & Paginate[Order] & DS đơn hàng buyer \\
ListSellerConfirmed & SellerID, Status[], Page & Paginate[Order] & DS đơn hàng seller \\
PayBuyerOrders   & OrderIDs, PaymentOption & Payment, RedirectUrl & Thanh toán nhiều đơn \\
ConfirmPayment   & RefID, Status & -- & Callback từ payment gateway \\
\addlinespace
GetSellerOrderStats & SellerID, DateRange & SellerOrderStats & Thống kê tổng hợp \\
GetSellerOrderTimeSeries & SellerID, DateRange & []TimeSeriesPoint & Biểu đồ theo thời gian \\
GetSellerPendingActions & SellerID & PendingActions & Số lượng hành động chờ \\
GetSellerTopProducts & SellerID, Limit & []TopProduct & Top SP bán chạy \\
\bottomrule
\end{tabular}
\end{table}
