# Order Restate Workflow Refactor — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development` (recommended) or `superpowers:executing-plans` to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace `BuyerCheckout` + `ConfirmSellerPending` Service handlers with three Restate Workflows (`CheckoutWorkflow`, `ConfirmWorkflow`, `PayoutWorkflow`) backed by a shared LIFO saga compensator helper. Refund schema migrated to per-order. Webhook handlers funneled through new `OnPaymentResult` biz indirection.

**Architecture:** Each workflow owns one phase of the order lifecycle. Workflow IDs are pre-allocated by the HTTP layer (`checkout_session_id`, `confirm_session_id`, `order_id`). HTTP submits via `Send` then attaches via `Request("WaitPaymentURL")` for sync URL return. Webhook gateway pushes results through `OrderHandler.OnPaymentResult` → workflow shared handler `ResolvePayment`. Refund handlers signal `PayoutWorkflow.OnRefundChanged` on every state change; `PayoutWorkflow` watches DB snapshot, releases or cancels payout.

**Tech Stack:** Go, `github.com/restatedev/sdk-go v0.24.0`, sqlc, pgx/v5, Echo (HTTP), pgtempl (codegen).

**Reference spec:** `docs/superpowers/specs/2026-04-28-order-restate-workflow-design.md`. **Testing tasks omitted** per user instruction.

- [ ] **Step 1: Create the helper file**

```go
package orderbiz

import (
	"log/slog"

	restate "github.com/restatedev/sdk-go"
)

// Saga collects compensators in append order and runs them LIFO on failure.
// Each compensator is wrapped in restate.RunVoid → Restate retries idempotently
// on failure. The compensators slice lives in Go memory; on workflow replay it
// is rebuilt deterministically because every action between Defer() calls is
// itself wrapped in restate.Run, so journal replay restores the slice to the
// exact state it had at the crash point.
type Saga struct {
	ctx          restate.WorkflowContext
	compensators []sagaStep
}

type sagaStep struct {
	name string
	fn   func(restate.RunContext) error
}

// NewSaga creates an empty saga bound to the given workflow context.
func NewSaga(ctx restate.WorkflowContext) *Saga {
	return &Saga{ctx: ctx}
}

// Defer appends a compensator. Call BEFORE performing the action it compensates.
func (s *Saga) Defer(name string, fn func(restate.RunContext) error) {
	s.compensators = append(s.compensators, sagaStep{name: name, fn: fn})
}

// Compensate runs all deferred compensators LIFO. Each runs in restate.RunVoid
// so Restate retries indefinitely on failure (compensators must be idempotent).
func (s *Saga) Compensate() {
	for i := len(s.compensators) - 1; i >= 0; i-- {
		step := s.compensators[i]
		if err := restate.RunVoid(s.ctx, func(rctx restate.RunContext) error {
			return step.fn(rctx)
		}, restate.WithName("compensate:"+step.name)); err != nil {
			slog.Error("saga compensate", slog.String("step", step.name), slog.Any("error", err))
		}
	}
}

// Clear drops all pending compensators. Call on the success exit path.
func (s *Saga) Clear() { s.compensators = nil }
```

	return strconv.FormatInt(sessionID, 10)

// workflowForSession maps payment_session.kind to (workflowName, workflowID).

func NewCheckoutWorkflowHandler(base *OrderHandler) *CheckoutWorkflowHandler {
	return &CheckoutWorkflowHandler{OrderHandler: base}

func (h *CheckoutWorkflowHandler) ServiceName() string { return "CheckoutWorkflow" }

func (h *CheckoutWorkflowHandler) Run(
	ctx restate.WorkflowContext,
	input CheckoutWorkflowInput,
) (out CheckoutWorkflowOutput, err error) {
	defer metrics.TrackHandler("checkout_workflow", "Run", &err)()

	if err = validator.Validate(input); err != nil {
		return out, sharedmodel.WrapErr("validate checkout", err)
	}

	saga := NewSaga(ctx)
	var (
		cancelled bool
		expired   bool
	)
	defer func() {
		if err != nil || cancelled || expired {
			saga.Compensate()
			if !errors.Is(err, nil) {
				err = restate.TerminalError(err)
			}
		}
	}()

	// TODO Step 3+: port body from checkout.go BuyerCheckout, wrapping side effects with saga.Defer.

	return out, nil
}

func (h *CheckoutWorkflowHandler) WaitPaymentURL(ctx restate.WorkflowSharedContext, _ struct{}) (string, error) {
	return restate.Promise[string](ctx, "payment_url").Result()
}

func (h *CheckoutWorkflowHandler) ResolvePayment(ctx restate.WorkflowSharedContext, ev PaymentEvent) error {
	return restate.Promise[PaymentEvent](ctx, "payment_event").Resolve(ev)
}

func (h *CheckoutWorkflowHandler) CancelCheckout(ctx restate.WorkflowSharedContext, _ struct{}) error {

	restate "github.com/restatedev/sdk-go"

	accountmodel "shopnexus-server/internal/module/account/model"

	"github.com/guregu/null/v6"

const escrowHold = 7 * 24 * time.Hour

type PayoutWorkflowHandler struct {
	*OrderHandler
}

func NewPayoutWorkflowHandler(base *OrderHandler) *PayoutWorkflowHandler {
	return &PayoutWorkflowHandler{OrderHandler: base}
}

func (h *PayoutWorkflowHandler) ServiceName() string { return "PayoutWorkflow" }

func (h *PayoutWorkflowHandler) Run(
	ctx restate.WorkflowContext,
	input PayoutInput,
) (out PayoutOutput, err error) {
	out.OrderID = input.OrderID

	// Step 1: create payout session + payout tx (Pending). Idempotent on replay.
	payoutSessionID, err := restate.Run(ctx, func(rctx restate.RunContext) (int64, error) {
		ps, e := h.storage.Querier().CreateDefaultPaymentSession(rctx, orderdb.CreateDefaultPaymentSessionParams{
			Kind:        SessionKindSellerPayout,
			Status:      orderdb.OrderStatusPending,
			FromID:      uuid.NullUUID{},
			ToID:        uuid.NullUUID{UUID: input.SellerID, Valid: true},
			Note:        "seller payout (escrow)",
			Currency:    input.Currency,
			TotalAmount: input.PaidTotal,
			Data:        json.RawMessage("{}"),
			DatePaid:    null.Time{},
			DateExpired: time.Now().Add(365 * 24 * time.Hour),
		})
		if e != nil { return 0, e }
		_, e = h.storage.Querier().CreateDefaultTransaction(rctx, orderdb.CreateDefaultTransactionParams{
			SessionID:    ps.ID,
			Status:       orderdb.OrderStatusPending,
			Note:         "seller payout (escrow)",
			Amount:       input.PaidTotal,
			FromCurrency: input.Currency,
			ToCurrency:   input.Currency,
			ExchangeRate: mustNumericOne(),
			Data:         json.RawMessage("{}"),
		})
		return ps.ID, e
	})
	if err != nil { return out, err }

	deadline := time.Now().Add(escrowHold)
	iter := 0

	for {
		// Reload snapshot fresh each iteration.
		snap, err := restate.Run(ctx, func(rctx restate.RunContext) (RefundSnapshot, error) {
			row, e := h.storage.Querier().GetRefundSnapshotByOrder(rctx, input.OrderID)
			return RefundSnapshot{
				HasActiveRefund:    row.HasActiveRefund,
				LastRefundApproved: row.LastRefundApproved,
			}, e
		})
		if err != nil { return out, err }

		switch {
		case snap.LastRefundApproved:
			if err = restate.RunVoid(ctx, func(rctx restate.RunContext) error {
				_, e := h.storage.Querier().MarkPaymentSessionCancelled(rctx, payoutSessionID)
				return e
			}); err != nil { return out, err }
			notifySellerPayoutCancelled(ctx, h, input.SellerID, input.OrderID)
			out.Outcome = "refunded"
			return out, nil

		case !snap.HasActiveRefund && time.Now().After(deadline):
			if err = restate.RunVoid(ctx, func(rctx restate.RunContext) error {
				return h.releasePayout(rctx, payoutSessionID, input)
			}); err != nil { return out, err }
			notifySellerPayoutReleased(ctx, h, input.SellerID, input.OrderID)
			out.Outcome = "released"
			return out, nil
		}

		// Wait for next event or deadline.
		iter++
		if err = restate.Set(ctx, "refund_iter", iter); err != nil { return out, err }
		signalFut := restate.Promise[any](ctx, fmt.Sprintf("refund_changed_%d", iter)).Future()
		var futs []restate.Future
		futs = append(futs, signalFut)
		if remaining := time.Until(deadline); remaining > 0 {
			futs = append(futs, restate.After(ctx, remaining).Future())
		}
		_, _ = restate.WaitFirst(ctx, futs...)
		// loop reevaluates
	}
}

func (h *PayoutWorkflowHandler) OnRefundChanged(ctx restate.WorkflowSharedContext, _ struct{}) error {
	iter, _ := restate.Get[int](ctx, "refund_iter")
	return restate.Promise[any](ctx, fmt.Sprintf("refund_changed_%d", iter)).Resolve(struct{}{})
}

// releasePayout marks the payout session Success, marks the payout tx Success,
// and credits the seller's wallet via the shared CreditFromSession helper.
func (h *OrderHandler) releasePayout(rctx restate.RunContext, payoutSessionID int64, input PayoutInput) error {
	if _, e := h.storage.Querier().MarkPaymentSessionSuccess(rctx, payoutSessionID); e != nil {
		return e
	}
	// Find the single payout tx in the session and mark it Success.
	txs, e := h.storage.Querier().ListTransactionsBySession(rctx, payoutSessionID)
	if e != nil { return e }
	for _, tx := range txs {
		if tx.Status == orderdb.OrderStatusPending && tx.Amount > 0 {
			if _, e := h.storage.Querier().MarkTransactionSuccess(rctx, tx.ID); e != nil { return e }
		}
	}
	if _, e := h.account.WalletCredit(rctx, accountbiz.WalletCreditParams{
		AccountID: input.SellerID,
		Amount:    input.PaidTotal,
		Reference: fmt.Sprintf("payout-session:%d", payoutSessionID),
		Note:      "escrow released",
	}); e != nil { return e }
	return nil
}

func notifySellerPayoutReleased(ctx restate.WorkflowContext, h *PayoutWorkflowHandler, sellerID, orderID uuid.UUID) {
