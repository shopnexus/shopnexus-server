# Order Lifecycle Workflow — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the scattered DB-status-machine pattern (timeout sweeps, scheduled escrow, auto-promote, item-cancel ad hoc) with a single Restate `Workflow` per `payment_session`. The workflow's `Run` function becomes the source of truth for order lifecycle phase transitions.

**Architecture:** New `OrderWorkflow` struct in `internal/module/order/biz/order_workflow.go`, holding `*OrderHandler`. Workflow ID = `payment_session.id`. Five phases: pay → seller decide → deliver → escrow vs refund → payout. Forward terminal helpers (no saga) for Phase 1 timeout & Phase 2 reject. Existing service handlers shrink to thin signal forwarders. Rolled out across 5 PRs per spec §11; each PR independently revertable, build-green.

**Tech Stack:** Go 1.x, Restate Go SDK v0.24.0, sqlc + pgtempl for query generation, uber/fx for DI, Postgres 16.

**Spec:** `docs/superpowers/specs/2026-04-27-order-lifecycle-workflow-design.md`

**Backend working directory:** `/home/beanbocchi/shopnexus/shopnexus-server` (referenced as `$BE` below).

```go
// Run is the workflow's main state machine. Forward-only: each terminal branch
// calls a forward helper directly. No saga slice, no defer rollback (see spec
// §6 for rationale).
//
// Phase sequence:
//   1. Wait for buyer to pay (timer: paymentExpiry).
//   2. Wait for seller decision (confirm with orderID + payoutSessionID, or reject).
//   3. Wait for delivery signal.
//   4. Race escrow timer vs refund-raised signal.
//   5. Release payout to seller (terminal act).
func (w *OrderWorkflow) Run(
	ctx restate.WorkflowContext,
	params StartOrderWorkflowParams,
) error {
	// ── PHASE 1: payment ──────────────────────────────────
	paid := restate.Promise[bool](ctx, promPaid)
	if winner := restate.WaitFirst(ctx, paid, restate.After(ctx, workflowPaymentExpiry)); winner != paid {
		return w.tearDownSession(ctx, params, "payment timeout")
	}

	// ── PHASE 2: seller decision ──────────────────────────
	decision, err := restate.Promise[SellerDecisionSignal](ctx, promDecision).Result()
	if err != nil {
		return sharedmodel.WrapErr("await seller decision", err)
	}
	if !decision.Confirm {
		return w.tearDownSession(ctx, params, "seller rejected: "+decision.Reason)
	}

	// ── PHASE 3: delivered ────────────────────────────────
	if _, err := restate.Promise[bool](ctx, promDelivered).Result(); err != nil {
		return sharedmodel.WrapErr("await delivered", err)
	}

	// ── PHASE 4: escrow vs refund ─────────────────────────
	refundRaised := restate.Promise[bool](ctx, promRefundRaised)
	if winner := restate.WaitFirst(ctx, refundRaised, restate.After(ctx, workflowEscrowWindow)); winner == refundRaised {
		resolution, err := restate.Promise[RefundResolution](ctx, promRefundResolved).Result()
		if err != nil {
			return sharedmodel.WrapErr("await refund resolution", err)
		}
		if resolution.Approved {
			return nil // refund flow already credited buyer + cancelled payout
		}
		// rejected → fall through to payout
	}

	// ── PHASE 5: payout (workflow's only act) ─────────────
	return w.releasePayout(ctx, decision.OrderID, decision.PayoutSessionID)
}
```

```go
// tearDownSession transitions the session and its item to a terminal state.
// Called by Phase 1 timeout (session was Pending) and Phase 2 reject (session
// may be Success after payment confirmed). Idempotent.
//
// Cross-module calls (CreditFromSession, ReleaseInventory) rely on Restate's
// exactly-once journal — they are invoked at most once per workflow because
// workflow_id == session_id is unique and teardown is a terminal branch
// (see spec §10.2).
func (w *OrderWorkflow) tearDownSession(
	ctx restate.Context,
	p StartOrderWorkflowParams,
	reason string,
) error {
	// One atomic Run for DB writes (status terminate + item cancel).
	item, err := restate.Run(ctx, func(rctx restate.RunContext) (orderdb.OrderItem, error) {
		if _, e := w.h.storage.Querier().MarkPaymentSessionTerminated(rctx, p.SessionID); e != nil && !errors.Is(e, pgx.ErrNoRows) {
			return orderdb.OrderItem{}, sharedmodel.WrapErr("mark session terminated", e)
		}
		it, e := w.h.storage.Querier().CancelItem(rctx, orderdb.CancelItemParams{
			ID:            p.ItemID,
			CancelledByID: uuid.NullUUID{UUID: p.BuyerID, Valid: true},
		})
		if e != nil && !errors.Is(e, pgx.ErrNoRows) {
			return orderdb.OrderItem{}, sharedmodel.WrapErr("cancel item", e)
		}
		return it, nil
	}, restate.WithName("tearDown.dbWrites"))
	if err != nil {
		return err
	}

	// Refund settled portion (CreditFromSession sums positive Success txs).
	// If session was Pending and no leg settled, this no-ops (sum == 0).
	if _, err = w.h.CreditFromSession(ctx, CreditFromSessionParams{
		SessionID:  p.SessionID,
		AccountID:  p.BuyerID,
		CreditType: "Refund",
		Reference:  fmt.Sprintf("wf-teardown:%d", p.SessionID),
		Note:       reason,
	}); err != nil {
		return sharedmodel.WrapErr("credit buyer (teardown)", err)
	}

	// Release inventory back to available stock.
	return w.h.inventory.ReleaseInventory(ctx, inventorybiz.ReleaseInventoryParams{
		Items: []inventorybiz.ReleaseInventoryItem{{
			RefType: inventorydb.InventoryStockRefTypeProductSku,
			RefID:   item.SkuID,
			Amount:  item.Quantity,
		}},
	})
}
```

```go
// releasePayout transitions the payout session to Success and credits the
// seller's wallet. Called from Phase 5 (the only terminal happy path).
//
// Mirrors the body of OrderHandler.ReleaseEscrow, minus the now-redundant
// HasActiveRefundForOrder check (workflow guarantees no refund raised — Phase 4
// would have branched).
func (w *OrderWorkflow) releasePayout(
	ctx restate.Context,
	orderID uuid.UUID,
	payoutSessionID int64,
) error {
	// Find the pending payout tx in the session.
	pendingTxID, err := restate.Run(ctx, func(rctx restate.RunContext) (int64, error) {
		txs, e := w.h.storage.Querier().ListTransactionsBySession(rctx, payoutSessionID)
		if e != nil {
			return 0, sharedmodel.WrapErr("list payout session txs", e)
		}
		for _, tx := range txs {
			if tx.Status == orderdb.OrderStatusPending && tx.Amount > 0 && !tx.ReversesID.Valid {
				return tx.ID, nil
			}
		}
		return 0, ordermodel.ErrTxNotFound.Terminal()
	})
	if err != nil {
		return err
	}

	// Mark Success — auto-promotes session via MarkTxSuccess auto-promote logic.
	if err := w.h.MarkTxSuccess(ctx, MarkTxSuccessParams{TxID: pendingTxID, DateAt: time.Now()}); err != nil {
		return sharedmodel.WrapErr("mark payout tx success", err)
	}

	// Load order to get seller ID.
	order, err := restate.Run(ctx, func(rctx restate.RunContext) (orderdb.OrderOrder, error) {
		return w.h.storage.Querier().GetOrder(rctx, uuid.NullUUID{UUID: orderID, Valid: true})
	})
	if err != nil {
		return sharedmodel.WrapErr("get order", err)
	}

	// Credit seller's wallet via session sum (canonical helper).
	if _, err := w.h.CreditFromSession(ctx, CreditFromSessionParams{
		SessionID:  payoutSessionID,
		AccountID:  order.SellerID,
		CreditType: "Payout",
		Reference:  fmt.Sprintf("wf-payout:%d", payoutSessionID),
		Note:       "escrow release",
	}); err != nil {
		return sharedmodel.WrapErr("credit seller (payout)", err)
	}
	return nil
}
```

```go
// --- Signal handlers ---
// Each resolves one named Promise. Idempotent at the workflow level via
// Promise's single-resolve semantic (see spec §7 Q4 — verify behavior at
// runtime smoke; add a guard if Resolve double-call errors).

// ConfirmPayment is called by MarkTxSuccess (or BuyerCheckout for pure-wallet
// sessions) when all payment txs in the session have settled Success.
func (w *OrderWorkflow) ConfirmPayment(ctx restate.WorkflowSharedContext) error {
	return restate.Promise[bool](ctx, promPaid).Resolve(true)
}

// SellerDecide is called by ConfirmSellerPending (Confirm=true with orderID +
// payoutSessionID) and RejectSellerPending (Confirm=false with reason).
func (w *OrderWorkflow) SellerDecide(
	ctx restate.WorkflowSharedContext,
	d SellerDecisionSignal,
) error {
	return restate.Promise[SellerDecisionSignal](ctx, promDecision).Resolve(d)
}

// MarkDelivered is called by the transport-status webhook when the transport
// flips to Delivered. Fan-out happens at the call site (one signal per item's
// session).
func (w *OrderWorkflow) MarkDelivered(ctx restate.WorkflowSharedContext) error {
	return restate.Promise[bool](ctx, promDelivered).Resolve(true)
}

// RaiseRefund is called by CreateBuyerRefund after the refund row + return
// transport are created.
func (w *OrderWorkflow) RaiseRefund(ctx restate.WorkflowSharedContext) error {
	return restate.Promise[bool](ctx, promRefundRaised).Resolve(true)
}

// ResolveRefund is called by ApproveRefundStage2 (Approved=true, with RefundID)
// or RejectRefund (Approved=false).
func (w *OrderWorkflow) ResolveRefund(
	ctx restate.WorkflowSharedContext,
	r RefundResolution,
) error {
	return restate.Promise[RefundResolution](ctx, promRefundResolved).Resolve(r)
}
```
