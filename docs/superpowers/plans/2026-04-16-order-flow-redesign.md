# Order Flow Redesign: Pay-First Checkout — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Rewrite the order checkout flow so buyers pay at checkout (before seller confirmation), with wallet support and platform-owned shipping.

**Architecture:** Big-bang rewrite of checkout.go, incoming.go, pay.go, and order.go. New wallet tables in account schema. Item status inferred from payment.status → order_id → transport.status instead of an explicit status column. Restate delayed calls for payment timeout (15 min) and seller confirm timeout (48h).

**Tech Stack:** Go 1.26, Restate SDK, PostgreSQL 18, SQLC, Uber fx, Echo v4

**Spec:** `docs/superpowers/specs/2026-04-16-order-flow-redesign-design.md`

```sql
-- name: GetWallet :one
SELECT * FROM account.wallet WHERE account_id = $1;

-- name: CreateWallet :one
INSERT INTO account.wallet (account_id, balance)
VALUES ($1, 0)
RETURNING *;

-- name: DebitWallet :one
-- Atomically debit up to @amount from wallet. Returns updated wallet.
-- If balance < amount, deducts whatever is available (partial debit).
UPDATE account.wallet
SET balance = CASE
    WHEN balance >= @amount THEN balance - @amount
    ELSE 0
END
WHERE account_id = @account_id
RETURNING *;

-- name: CreditWallet :one
-- Atomically credit amount to wallet. Creates wallet if not exists (upsert).
INSERT INTO account.wallet (account_id, balance)
VALUES (@account_id, @amount)
ON CONFLICT (account_id) DO UPDATE
SET balance = account.wallet.balance + @amount
RETURNING *;

-- name: CreateWalletTransaction :one
INSERT INTO account.wallet_transaction (account_id, type, amount, reference_id, note)
VALUES ($1, $2, $3, $4, $5)
RETURNING *;

-- name: ListWalletTransactions :many
SELECT * FROM account.wallet_transaction
WHERE account_id = $1
ORDER BY date_created DESC
LIMIT $2 OFFSET $3;
```

```go
package accountbiz

import (
	"fmt"

	accountdb "shopnexus-server/internal/module/account/db/sqlc"

	"github.com/google/uuid"
	"github.com/restatedev/sdk-go/restate"
)

func (b *AccountHandler) GetWalletBalance(ctx restate.Context, accountID uuid.UUID) (int64, error) {
	type result struct {
		Balance int64 `json:"balance"`
	}
	r, err := restate.Run(ctx, func(ctx restate.RunContext) (result, error) {
		wallet, err := b.storage.Querier().GetWallet(ctx, accountID)
		if err != nil {
			// No wallet yet — balance is 0
			return result{Balance: 0}, nil
		}
		return result{Balance: wallet.Balance}, nil
	})
	if err != nil {
		return 0, err
	}
	return r.Balance, nil
}

func (b *AccountHandler) WalletDebit(ctx restate.Context, params WalletDebitParams) (_ WalletDebitResult, err error) {
	type result struct {
		Deducted int64 `json:"deducted"`
		Balance  int64 `json:"balance"`
	}

	r, err := restate.Run(ctx, func(ctx restate.RunContext) (result, error) {
		// Get current balance first to calculate actual deduction
		wallet, err := b.storage.Querier().GetWallet(ctx, params.AccountID)
		if err != nil {
			// No wallet = nothing to debit
			return result{Deducted: 0, Balance: 0}, nil
		}

		if wallet.Balance == 0 {
			return result{Deducted: 0, Balance: 0}, nil
		}

		actualDeduct := params.Amount
		if wallet.Balance < params.Amount {
			actualDeduct = wallet.Balance
		}

		updated, err := b.storage.Querier().DebitWallet(ctx, accountdb.DebitWalletParams{
			AccountID: params.AccountID,
			Amount:    actualDeduct,
		})
		if err != nil {
			return result{}, fmt.Errorf("debit wallet: %w", err)
		}

		_, err = b.storage.Querier().CreateWalletTransaction(ctx, accountdb.CreateWalletTransactionParams{
			AccountID:   params.AccountID,
			Type:        accountdb.WalletTransactionTypePayment,
			Amount:      actualDeduct,
			ReferenceID: &params.Reference,
			Note:        &params.Note,
		})
		if err != nil {
			return result{}, fmt.Errorf("create wallet tx: %w", err)
		}

		return result{Deducted: actualDeduct, Balance: updated.Balance}, nil
	})
	if err != nil {
		return WalletDebitResult{}, err
	}
	return WalletDebitResult{Deducted: r.Deducted, Balance: r.Balance}, nil
}

func (b *AccountHandler) WalletCredit(ctx restate.Context, params WalletCreditParams) error {
	return restate.RunVoid(ctx, func(ctx restate.RunContext) error {
		txType := accountdb.WalletTransactionType(params.Type)
		if !txType.Valid() {
			return fmt.Errorf("invalid wallet transaction type: %s", params.Type)
		}

		_, err := b.storage.Querier().CreditWallet(ctx, accountdb.CreditWalletParams{
			AccountID: params.AccountID,
			Amount:    params.Amount,
		})
		if err != nil {
			return fmt.Errorf("credit wallet: %w", err)
		}

		_, err = b.storage.Querier().CreateWalletTransaction(ctx, accountdb.CreateWalletTransactionParams{
			AccountID:   params.AccountID,
			Type:        txType,
			Amount:      params.Amount,
			ReferenceID: &params.Reference,
			Note:        &params.Note,
		})
		if err != nil {
			return fmt.Errorf("create wallet tx: %w", err)
		}
		return nil
	})
}
```

**Files:**

- [ ] **Step 1: Rewrite BuyerCheckout**

Replace the entire `BuyerCheckout` function body. The new flow:

```go
func (b *OrderHandler) BuyerCheckout(
	ctx restate.Context,
	params BuyerCheckoutParams,
) (_ BuyerCheckoutResult, err error) {
	defer metrics.TrackHandler("order", "BuyerCheckout", &err)()

	if err := sharedmodel.Validate(params); err != nil {
		return BuyerCheckoutResult{}, err
	}

	// Step 1: Fetch products from catalog
	skuIDs := make([]uuid.UUID, len(params.Items))
	for i, item := range params.Items {
		skuIDs[i] = item.SkuID
	}

	skuMap, err := b.catalog.ListProductSku(ctx, catalogbiz.ListProductSkuParams{SkuIDs: skuIDs})
	if err != nil {
		return BuyerCheckoutResult{}, fmt.Errorf("fetch SKUs: %w", err)
	}

	spuIDs := make([]uuid.UUID, 0)
	for _, sku := range skuMap {
		spuIDs = append(spuIDs, sku.SpuID)
	}
	spuMap, err := b.catalog.ListProductSpu(ctx, catalogbiz.ListProductSpuParams{SpuIDs: spuIDs})
	if err != nil {
		return BuyerCheckoutResult{}, fmt.Errorf("fetch SPUs: %w", err)
	}

	// Step 2: Reserve inventory per item
	type reserveInfo struct {
		SkuID     uuid.UUID
		Quantity  int64
		SerialIDs json.RawMessage
	}
	reserves := make([]reserveInfo, 0, len(params.Items))
	for _, item := range params.Items {
		result, err := b.inventory.ReserveInventory(ctx, inventorybiz.ReserveInventoryParams{
			SkuID:    item.SkuID,
			Quantity: item.Quantity,
		})
		if err != nil {
			return BuyerCheckoutResult{}, fmt.Errorf("reserve inventory for SKU %s: %w", item.SkuID, err)
		}
		serialBytes, _ := json.Marshal(result.SerialIDs)
		reserves = append(reserves, reserveInfo{
			SkuID: item.SkuID, Quantity: item.Quantity, SerialIDs: serialBytes,
		})
	}

	// Step 3: Quote transport per item individually
	type quoteInfo struct {
		Cost int64 `json:"cost"`
	}
	transportQuotes := make([]quoteInfo, len(params.Items))
	for i, item := range params.Items {
		sku := skuMap[item.SkuID]
		spu := spuMap[sku.SpuID]

		sellerContacts, err := b.account.GetDefaultContact(ctx, []uuid.UUID{spu.SellerID})
		if err != nil {
			return BuyerCheckoutResult{}, fmt.Errorf("get seller contact: %w", err)
		}
		sellerContact := sellerContacts[spu.SellerID]

		transportClient, err := b.getTransportClient(item.TransportOption)
		if err != nil {
			return BuyerCheckoutResult{}, err
		}

		q, err := restate.Run(ctx, func(ctx restate.RunContext) (quoteInfo, error) {
			result, err := transportClient.Quote(ctx, transport.QuoteParams{
				Items: []transport.ItemMetadata{{
					SkuID:    item.SkuID,
					Quantity: item.Quantity,
				}},
				FromAddress: sellerContact.Address,
				ToAddress:   params.Address,
			})
			if err != nil {
				return quoteInfo{}, err
			}
			return quoteInfo{Cost: result.Cost}, nil
		})
		if err != nil {
			return BuyerCheckoutResult{}, fmt.Errorf("quote transport for SKU %s: %w", item.SkuID, err)
		}
		transportQuotes[i] = q
	}

	// Step 4: Calculate total
	var productTotal, transportTotal int64
	for i, item := range params.Items {
		sku := skuMap[item.SkuID]
		productTotal += int64(sku.Price) * item.Quantity
		transportTotal += transportQuotes[i].Cost
	}
	total := productTotal + transportTotal

	// Step 5: Process wallet deduction (if opted in)
	var walletDeducted int64
	if params.UseWallet {
		walletResult, err := b.account.WalletDebit(ctx, accountbiz.WalletDebitParams{
			AccountID: params.Account.ID,
			Amount:    total,
			Reference: fmt.Sprintf("checkout:%s", params.Account.ID),
			Note:      "Checkout payment",
		})
		if err != nil {
			return BuyerCheckoutResult{}, fmt.Errorf("wallet debit: %w", err)
		}
		walletDeducted = walletResult.Deducted
	}
	remaining := total - walletDeducted

	// Step 6: Create payment
	type payResult struct {
		PaymentID   int64  `json:"payment_id"`
		RedirectURL string `json:"redirect_url"`
	}

	var payment payResult
	if remaining > 0 {
		// Need payment provider
		paymentClient, err := b.getPaymentClient(params.PaymentOption)
		if err != nil {
			return BuyerCheckoutResult{}, err
		}

		payment, err = restate.Run(ctx, func(ctx restate.RunContext) (payResult, error) {
			// Create payment record
			dbPayment, err := b.storage.Querier().CreateDefaultPayment(ctx, orderdb.CreateDefaultPaymentParams{
				AccountID:  params.Account.ID,
				Option:     params.PaymentOption,
				Amount:     remaining,
				Data:       json.RawMessage(`{}`),
				DateExpired: time.Now().Add(15 * time.Minute),
			})
			if err != nil {
				return payResult{}, fmt.Errorf("create payment: %w", err)
			}

			// Call payment provider
			providerResult, err := paymentClient.Create(ctx, payment.CreateParams{
				RefID:       fmt.Sprintf("%d", dbPayment.ID),
				Amount:      remaining,
				Description: "ShopNexus Checkout",
			})
			if err != nil {
				return payResult{}, fmt.Errorf("payment provider: %w", err)
			}

			return payResult{
				PaymentID:   dbPayment.ID,
				RedirectURL: providerResult.RedirectURL,
			}, nil
		})
		if err != nil {
			return BuyerCheckoutResult{}, err
		}
	} else {
		// Wallet covers everything — create payment with status Success
		payment, err = restate.Run(ctx, func(ctx restate.RunContext) (payResult, error) {
			now := time.Now()
			dbPayment, err := b.storage.Querier().CreatePayment(ctx, orderdb.CreatePaymentParams{
				AccountID:  params.Account.ID,
				Option:     "wallet",
				Status:     orderdb.StatusSuccess,
				Amount:     total,
				Data:       json.RawMessage(`{}`),
				DatePaid:   &now,
				DateExpired: now.Add(24 * time.Hour), // doesn't matter, already paid
			})
			if err != nil {
				return payResult{}, fmt.Errorf("create wallet payment: %w", err)
			}
			return payResult{PaymentID: dbPayment.ID}, nil
		})
		if err != nil {
			return BuyerCheckoutResult{}, err
		}
	}

	// Step 7: Create pending items (linked to payment)
	type createdItem struct {
		ID int64 `json:"id"`
	}
	createdItems, err := restate.Run(ctx, func(ctx restate.RunContext) ([]createdItem, error) {
		items := make([]createdItem, 0, len(params.Items))
		for i, item := range params.Items {
			sku := skuMap[item.SkuID]
			spu := spuMap[sku.SpuID]
			skuName := buildSkuName(spu, sku)
			paidAmount := int64(sku.Price) * item.Quantity

			dbItem, err := b.storage.Querier().CreatePendingItem(ctx, orderdb.CreatePendingItemParams{
				AccountID:             params.Account.ID,
				SellerID:              spu.SellerID,
				Address:               params.Address,
				SkuID:                 item.SkuID,
				SkuName:               skuName,
				Quantity:              item.Quantity,
				UnitPrice:             int64(sku.Price),
				PaidAmount:            paidAmount,
				SerialIds:             reserves[i].SerialIDs,
				Note:                  nilIfEmpty(item.Note),
				TransportOption:       &item.TransportOption,
				TransportCostEstimate: &transportQuotes[i].Cost,
				PaymentID:             &payment.PaymentID,
			})
			if err != nil {
				return nil, fmt.Errorf("create item: %w", err)
			}
			items = append(items, createdItem{ID: dbItem.ID})
		}
		return items, nil
	})
	if err != nil {
		return BuyerCheckoutResult{}, err
	}

	// Step 8: Start timeout / remove cart / notify
	if remaining > 0 {
		// Payment pending — start 15 min timeout
		restate.ServiceSend(ctx, "Order", "CancelUnpaidCheckout").
			Send(payment.PaymentID, restate.WithDelay(15*time.Minute))
	} else {
		// Wallet-only (already paid) — start 48h seller timeout
		restate.ServiceSend(ctx, "Order", "AutoCancelPendingItems").
			Send(payment.PaymentID, restate.WithDelay(48*time.Hour))
	}

	// Remove from cart if not BuyNow
	if !params.BuyNow {
		restate.RunVoid(ctx, func(ctx restate.RunContext) error {
			return b.storage.Querier().RemoveCheckoutItem(ctx, orderdb.RemoveCheckoutItemParams{
				AccountID: params.Account.ID,
				SkuIds:    skuIDs,
			})
		})
	}

	// Fire-and-forget notifications to sellers
	// (group items by seller, send one notification per seller)
	sellerItems := make(map[uuid.UUID][]string)
	for i, item := range params.Items {
		sku := skuMap[item.SkuID]
		spu := spuMap[sku.SpuID]
		_ = createdItems[i] // ensure index valid
		sellerItems[spu.SellerID] = append(sellerItems[spu.SellerID], spu.Name)
	}
	for sellerID, names := range sellerItems {
		summary := names[0]
		if len(names) > 1 {
			summary += fmt.Sprintf(" và %d sản phẩm khác", len(names)-1)
		}
		restate.ServiceSend(ctx, "Account", "CreateNotification").Send(accountbiz.CreateNotificationParams{
			AccountID: sellerID,
			Topic:     ordermodel.TopicOrderPending,
			Title:     "Đơn hàng mới",
			Body:      fmt.Sprintf("Bạn có đơn hàng mới: %s", summary),
		})
	}

	// Hydrate and return
	itemIDs := make([]int64, len(createdItems))
	for i, ci := range createdItems {
		itemIDs[i] = ci.ID
	}
	hydratedItems, err := b.hydrateItems(ctx, itemIDs)
	if err != nil {
		return BuyerCheckoutResult{}, err
	}

	var resultPayment *ordermodel.Payment
	var redirectURL *string
	if payment.PaymentID > 0 {
		p, err := b.fetchPayment(ctx, payment.PaymentID)
		if err == nil {
			resultPayment = &p
		}
	}
	if payment.RedirectURL != "" {
		redirectURL = &payment.RedirectURL
	}

	return BuyerCheckoutResult{
		Items:          hydratedItems,
		Payment:        resultPayment,
		RedirectUrl:    redirectURL,
		WalletDeducted: walletDeducted,
		Total:          total,
	}, nil
}
```

```go
// CancelUnpaidCheckout cancels items linked to an unpaid payment after 15 min timeout.
// Called by Restate delayed send.
func (b *OrderHandler) CancelUnpaidCheckout(ctx restate.Context, paymentID int64) (err error) {
	defer metrics.TrackHandler("order", "CancelUnpaidCheckout", &err)()

	// Fetch items linked to this payment that are still pending
	type itemInfo struct {
		ID                    int64     `json:"id"`
		AccountID             uuid.UUID `json:"account_id"`
		SkuID                 uuid.UUID `json:"sku_id"`
		Quantity              int64     `json:"quantity"`
		PaidAmount            int64     `json:"paid_amount"`
		TransportCostEstimate int64     `json:"transport_cost_estimate"`
	}

	items, err := restate.Run(ctx, func(ctx restate.RunContext) ([]itemInfo, error) {
		dbItems, err := b.storage.Querier().ListPendingPaymentItemsByPaymentID(ctx, &paymentID)
		if err != nil {
			return nil, fmt.Errorf("list items by payment: %w", err)
		}
		if len(dbItems) == 0 {
			return nil, nil // already confirmed or cancelled — skip
		}

		result := make([]itemInfo, len(dbItems))
		for i, di := range dbItems {
			result[i] = itemInfo{
				ID: di.ID, AccountID: di.AccountID, SkuID: di.SkuID,
				Quantity: di.Quantity, PaidAmount: di.PaidAmount,
				TransportCostEstimate: di.TransportCostEstimate,
			}
		}

		// Cancel the items
		ids := make([]int64, len(dbItems))
		for i, di := range dbItems {
			ids[i] = di.ID
		}
		_, err = b.storage.Querier().CancelItemsByIDs(ctx, ids)
		if err != nil {
			return nil, fmt.Errorf("cancel items: %w", err)
		}

		// Cancel the payment
		b.storage.Querier().UpdatePaymentStatus(ctx, orderdb.UpdatePaymentStatusParams{
			ID:     paymentID,
			Status: orderdb.StatusCancelled,
		})

		return result, nil
	})
	if err != nil {
		return err
	}

	if len(items) == 0 {
		return nil // nothing to cancel
	}

	// Release inventory per item
	for _, item := range items {
		b.inventory.ReleaseInventory(ctx, inventorybiz.ReleaseInventoryParams{
			SkuID:    item.SkuID,
			Quantity: item.Quantity,
		})
	}

	// Refund wallet (credit back whatever was deducted)
	// The wallet deduction amount is tracked at payment level, but we refund per-item
	// for traceability
	for _, item := range items {
		if item.PaidAmount+item.TransportCostEstimate > 0 {
			b.account.WalletCredit(ctx, accountbiz.WalletCreditParams{
				AccountID: item.AccountID,
				Amount:    item.PaidAmount + item.TransportCostEstimate,
				Type:      "Refund",
				Reference: fmt.Sprintf("payment_timeout:%d", paymentID),
				Note:      "Payment timeout — auto cancelled",
			})
		}
	}

	// Notify buyer
	restate.ServiceSend(ctx, "Account", "CreateNotification").Send(accountbiz.CreateNotificationParams{
		AccountID: items[0].AccountID,
		Topic:     ordermodel.TopicOrderCancelled,
		Title:     "Đơn hàng đã bị huỷ",
		Body:      "Đơn hàng đã bị huỷ do chưa thanh toán trong 15 phút.",
	})

	return nil
}

func (b *OrderHandler) AutoCancelPendingItems(ctx restate.Context, paymentID int64) (err error) {
	defer metrics.TrackHandler("order", "AutoCancelPendingItems", &err)()

	type itemInfo struct {
		ID                    int64     `json:"id"`
		AccountID             uuid.UUID `json:"account_id"`
		SellerID              uuid.UUID `json:"seller_id"`
		SkuID                 uuid.UUID `json:"sku_id"`
		Quantity              int64     `json:"quantity"`
		PaidAmount            int64     `json:"paid_amount"`
		TransportCostEstimate int64     `json:"transport_cost_estimate"`
	}

	items, err := restate.Run(ctx, func(ctx restate.RunContext) ([]itemInfo, error) {
		dbItems, err := b.storage.Querier().ListPendingPaymentItemsByPaymentID(ctx, &paymentID)
		if err != nil {
			return nil, fmt.Errorf("list items: %w", err)
		}

		// Filter: only items still pending (no order_id, not cancelled)
		var pending []itemInfo
		var pendingIDs []int64
		for _, di := range dbItems {
			if di.OrderID == nil && di.DateCancelled == nil {
				pending = append(pending, itemInfo{
					ID: di.ID, AccountID: di.AccountID, SellerID: di.SellerID,
					SkuID: di.SkuID, Quantity: di.Quantity,
					PaidAmount: di.PaidAmount, TransportCostEstimate: di.TransportCostEstimate,
				})
				pendingIDs = append(pendingIDs, di.ID)
			}
		}

		if len(pendingIDs) == 0 {
			return nil, nil
		}

		_, err = b.storage.Querier().CancelItemsByIDs(ctx, pendingIDs)
		if err != nil {
			return nil, fmt.Errorf("cancel items: %w", err)
		}
		return pending, nil
	})
	if err != nil {
		return err
	}

	if len(items) == 0 {
		return nil
	}

	// Release inventory + refund wallet per item
	for _, item := range items {
		b.inventory.ReleaseInventory(ctx, inventorybiz.ReleaseInventoryParams{
			SkuID:    item.SkuID,
			Quantity: item.Quantity,
		})
		b.account.WalletCredit(ctx, accountbiz.WalletCreditParams{
			AccountID: item.AccountID,
			Amount:    item.PaidAmount + item.TransportCostEstimate,
			Type:      "Refund",
			Reference: fmt.Sprintf("seller_timeout:%d", item.ID),
			Note:      "Seller did not confirm within 48 hours",
		})
	}

	// Notify buyer
	restate.ServiceSend(ctx, "Account", "CreateNotification").Send(accountbiz.CreateNotificationParams{
		AccountID: items[0].AccountID,
		Topic:     ordermodel.TopicOrderCancelled,
		Title:     "Đơn hàng đã bị huỷ tự động",
		Body:      "Seller chưa xác nhận đơn hàng trong 48 giờ. Tiền đã hoàn vào ví.",
	})

	// Notify sellers
	sellerMap := make(map[uuid.UUID]bool)
	for _, item := range items {
		if !sellerMap[item.SellerID] {
			sellerMap[item.SellerID] = true
			restate.ServiceSend(ctx, "Account", "CreateNotification").Send(accountbiz.CreateNotificationParams{
				AccountID: item.SellerID,
				Topic:     ordermodel.TopicOrderCancelled,
				Title:     "Đơn hàng đã bị huỷ tự động",
				Body:      "Bạn chưa xác nhận đơn hàng trong 48 giờ. Đơn đã bị huỷ.",
			})
		}
	}

	return nil
}
```

```go
	}

		Quantity              int64           `json:"quantity"`
		UnitPrice             int64           `json:"unit_price"`
		PaidAmount            int64           `json:"paid_amount"`
		Address               string          `json:"address"`
		TransportOption       string          `json:"transport_option"`
		TransportCostEstimate int64           `json:"transport_cost_estimate"`
		SerialIds             json.RawMessage `json:"serial_ids"`
	}

		Items   []fetchedItem `json:"items"`

		dbItems, err := b.storage.Querier().ListItem(ctx, orderdb.ListItemParams{
			IDs: params.ItemIDs,
		})
		if err != nil {
			return fetchResult{}, fmt.Errorf("fetch items: %w", err)
		}
		if len(dbItems) != len(params.ItemIDs) {
			return fetchResult{}, ordermodel.ErrItemNotFound
		}

		var buyerID uuid.UUID
		var transportOption string
		items := make([]fetchedItem, 0, len(dbItems))
		for i, di := range dbItems {

			if di.SellerID != params.Account.ID {
				return fetchResult{}, ordermodel.ErrNotOwner
			}
			if di.DateCancelled != nil {
				return fetchResult{}, ordermodel.ErrItemAlreadyCancelled
			}
			if di.OrderID != nil {
				return fetchResult{}, ordermodel.ErrItemAlreadyConfirmed
			}
			// All items must have same buyer
			if i == 0 {
				buyerID = di.AccountID
				transportOption = di.TransportOption
			} else {
				if di.AccountID != buyerID {
					return fetchResult{}, ordermodel.ErrDifferentBuyer
				}
				if di.TransportOption != transportOption {
					return fetchResult{}, ordermodel.ErrItemsTransportMismatch
				}
			}

			items = append(items, fetchedItem{
				ID: di.ID, AccountID: di.AccountID, SellerID: di.SellerID,
				SkuID: di.SkuID, SkuName: di.SkuName, Quantity: di.Quantity,
				UnitPrice: di.UnitPrice, PaidAmount: di.PaidAmount,
				Address: di.Address, TransportOption: di.TransportOption,
				TransportCostEstimate: di.TransportCostEstimate,
				SerialIds: di.SerialIds,
			})
		}
		return fetchResult{Items: items, BuyerID: buyerID}, nil
	})
	if err != nil {
		return ordermodel.Order{}, err
	}

	// Step 2: Get seller contact (from address)
	sellerContacts, err := b.account.GetDefaultContact(ctx, []uuid.UUID{params.Account.ID})
	if err != nil {
		return ordermodel.Order{}, fmt.Errorf("get seller contact: %w", err)
	}
	sellerContact := sellerContacts[params.Account.ID]

	transportClient, err := b.getTransportClient(fetched.Items[0].TransportOption)
	if err != nil {
		return ordermodel.Order{}, err
	}

	itemMeta := make([]transport.ItemMetadata, len(fetched.Items))
	for i, fi := range fetched.Items {
		itemMeta[i] = transport.ItemMetadata{
			SkuID:    fi.SkuID,
			Quantity: fi.Quantity,
		}
	}

	type transportResult struct {
		TransportID uuid.UUID `json:"transport_id"`
		Cost        int64     `json:"cost"`
	}
	tr, err := restate.Run(ctx, func(ctx restate.RunContext) (transportResult, error) {
		created, err := transportClient.Create(ctx, transport.CreateParams{
			Items:       itemMeta,
			FromAddress: sellerContact.Address,
			ToAddress:   fetched.Items[0].Address,
			Option:      fetched.Items[0].TransportOption,
		})
		if err != nil {
			return transportResult{}, fmt.Errorf("create transport: %w", err)
		}

		_, err = b.storage.Querier().CreateTransport(ctx, orderdb.CreateTransportParams{
			ID:     created.ID,
			Option: created.Option,
			Status: orderdb.TransportStatusPending,
			Cost:   created.Cost,
			Data:   created.Data,
		})
		if err != nil {
			return transportResult{}, fmt.Errorf("store transport: %w", err)
		}

		return transportResult{TransportID: created.ID, Cost: created.Cost}, nil
	})
	if err != nil {
		return ordermodel.Order{}, err
	}

	// Step 4: Create order and link items
	var productCost int64
	for _, fi := range fetched.Items {
		productCost += fi.UnitPrice * fi.Quantity
	}

	type orderResult struct {
		OrderID uuid.UUID `json:"order_id"`
	}
	or, err := restate.Run(ctx, func(ctx restate.RunContext) (orderResult, error) {
		order, err := b.storage.Querier().CreateOrder(ctx, orderdb.CreateOrderParams{
			BuyerID:         fetched.BuyerID,
			SellerID:        params.Account.ID,
			TransportID:     &tr.TransportID,
			ConfirmedByID:   &params.Account.ID,
			Address:         fetched.Items[0].Address,
			ProductCost:     productCost,
			ProductDiscount: 0, // buyer paid full at checkout
			TransportCost:   tr.Cost,
			Total:           productCost + tr.Cost,
			Note:            nilIfEmpty(params.Note),
			Data:            json.RawMessage(`{}`),
		})
		if err != nil {
			return orderResult{}, fmt.Errorf("create order: %w", err)
```

```go
func (b *OrderHandler) RejectSellerPending(ctx restate.Context, params RejectSellerPendingParams) (err error) {
	defer metrics.TrackHandler("order", "RejectSellerPending", &err)()

	if err := sharedmodel.Validate(params); err != nil {
		return err
	}

	type itemInfo struct {
		ID                    int64     `json:"id"`
		AccountID             uuid.UUID `json:"account_id"`
		SkuID                 uuid.UUID `json:"sku_id"`
		Quantity              int64     `json:"quantity"`
		PaidAmount            int64     `json:"paid_amount"`
		TransportCostEstimate int64     `json:"transport_cost_estimate"`
	}

	items, err := restate.Run(ctx, func(ctx restate.RunContext) ([]itemInfo, error) {
		dbItems, err := b.storage.Querier().ListItem(ctx, orderdb.ListItemParams{IDs: params.ItemIDs})
		if err != nil {
			return nil, err
		}
		if len(dbItems) != len(params.ItemIDs) {
			return nil, ordermodel.ErrItemNotFound
		}

		result := make([]itemInfo, 0, len(dbItems))
		for _, di := range dbItems {
			if di.SellerID != params.Account.ID {
				return nil, ordermodel.ErrNotOwner
			}
			if di.DateCancelled != nil {
				return nil, ordermodel.ErrItemAlreadyCancelled
			}
			if di.OrderID != nil {
				return nil, ordermodel.ErrItemAlreadyConfirmed
			}
			result = append(result, itemInfo{
				ID: di.ID, AccountID: di.AccountID, SkuID: di.SkuID,
				Quantity: di.Quantity, PaidAmount: di.PaidAmount,
				TransportCostEstimate: di.TransportCostEstimate,
			})
		}

		// Cancel items
		_, err = b.storage.Querier().CancelItemsByIDs(ctx, params.ItemIDs)
		if err != nil {
			return nil, fmt.Errorf("cancel items: %w", err)
		}
		return result, nil
	})
	if err != nil {
		return err
	}

	// Release inventory + refund wallet
	for _, item := range items {
		b.inventory.ReleaseInventory(ctx, inventorybiz.ReleaseInventoryParams{
			SkuID:    item.SkuID,
			Quantity: item.Quantity,
		})
		b.account.WalletCredit(ctx, accountbiz.WalletCreditParams{
			AccountID: item.AccountID,
			Amount:    item.PaidAmount + item.TransportCostEstimate,
			Type:      "Refund",
			Reference: fmt.Sprintf("seller_reject:%d", item.ID),
			Note:      "Seller rejected item",
		})
	}

	// Notify buyer
	restate.ServiceSend(ctx, "Account", "CreateNotification").Send(accountbiz.CreateNotificationParams{
		AccountID: items[0].AccountID,
		Topic:     ordermodel.TopicOrderCancelled,
		Title:     "Seller đã từ chối đơn hàng",
		Body:      "Tiền đã được hoàn vào ví của bạn.",
	})

	return nil
}
```
